94 double erfi(const double x) {
95 /* This function uses the Dawson Integral, i.e.
96 erfi(x) = 2 * Dawson * exp(x^2) / sqrt(pi)
97
98 The Dawson Integral is calculated by Stan Sykora's rational function approximations. URL : http://www.ebyte.it/library/codesnippets/DawsonIntegralApproximations.html (Dawson5())
99 */
100 // calculate the Dawson Integral:
101 double y,p,q;
102 y = x*x;
103 p = 1.0 + y*(0.1049934947 + y*(0.0424060604
104 + y*(0.0072644182 + y*(0.0005064034
105 + y*(0.0001789971)))));
106 q = 1.0 + y*(0.7715471019 + y*(0.2909738639
107 + y*(0.0694555761 + y*(0.0140005442
108 + y*(0.0008327945 + 2*0.0001789971*y)))));
109 double D = x*(p/q);
110
111 return D * std::exp(x*x) * M_2_SQRTPI; //
112 }

623 const Matrix linreg(
624 const std::vector<double>& xData,
625 const std::vector<double>& yData,
626 const int degree) {
627 // Create the matrix X
628 Matrix X(xData.size(), degree + 1);
629 for (size_t i = 0; i < xData.size(); i++) {
630 for (size_t j = 0; j < degree + 1; j++) {
631 X(i, j) = std::pow(xData[i], j);
632 }
633 }
634 // Create the matrix Y
635 Matrix Y(yData.size(), 1);
636 for (size_t i = 0; i < yData.size(); i++) {
637 Y(i, 0) = yData[i];
638 }
639 // Calculate the coefficients
640 Matrix X_T = X.T();
641 Matrix X_T_X = X_T * X;
642 Matrix X_T_X_inv = X_T_X.inv();
643 Matrix X_T_Y = X_T * Y;
644 Matrix coefficients = X_T_X_inv * X_T_Y;
645 return coefficients;
646 }
