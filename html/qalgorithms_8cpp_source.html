131 // Function to calculate the Designmatrix for a given a
132 void Peakmodel::calculateDesignMatrix(int a) {
133 // construct matrix
134 size_t m = 4;
135 size_t n = 2*a + 1;
136 Matrix X(n,m);
137
138 // x vector
139 std::vector<int> x(n);
140 for (size_t i = 0; i < n; i++)
141 {x[i] = i - a;}
142
143 // order vector
144 std::vector<int> p = {0, 1, 2, 2};
145
146 for (size_t i = 0; i < n; i++)
147 {
148 for (size_t j = 0; j < m; j++)
149 {
150 X(i, j) = std::pow(x[i],p[j]);
151 }
152 }
153
154 // modification of X due to p(2) and p(3)
155 for (size_t i = 0; i < n; i++)
156 {
157 if (x[i] < 0)
158 {X(i,3) = 0;}
159 else
160 {X(i,2) = 0;}
161 }
162 designMatrices.emplace(a, X);
163 }

331 void Peakmodel::gapFilling(Matrix& xyData) {
332 /* This function detects missing data and fills in the gaps. Specifically, when a gap is found, up to four zeros will be added to both sides of it. */
333 if (xyData.numRows() < 3) {
334 return;
335 }
336 // pre-define variables
337 std::vector<double> filledData;
338 std::vector<double> filledData_y;
339 bool skipPrevGapCalculation = false;
340 double lastPreviousGap = 0.0;
341 // add first
342 filledData.push_back(xyData(0,0));
343 filledData_y.push_back(xyData(0,1));
344 // find gaps
345 for (size_t i = 1; i < xyData.numRows() - 1; ++i) {
346 /* add current row from original data matrix to x and y vectors */
347 filledData.push_back(xyData(i,0));
348 filledData_y.push_back(xyData(i,1));
349 /* define current dinstance between next and current data point */
350 double gap = xyData(i + 1, 0) - xyData(i, 0);
351 /* define distance between current und former data point as reference*/
352 double previousGap = skipPrevGapCalculation ? lastPreviousGap : xyData(i, 0) - xyData(i - 1, 0);
353
354 lastPreviousGap = previousGap > lastPreviousGap ? previousGap : lastPreviousGap;
355 /* reset skip criterion */
356 skipPrevGapCalculation = false;
357
358 /* calculate the number of datapoints that should included into the gap */
359 int fillers = static_cast<int>(gap / previousGap) - 1;
360
361 /* define maximum number of fillers, it is 8 due to statistical conventions that a gap of more than 8 points is seen as full separation of the profiles, i.e., here a split can be performed later.*/
362 int maxFillers = 8;
363 int fillersToAdd = std::min(fillers, maxFillers);
364 // std::cout << gap << ", " << previousGap << ", " << fillersToAdd << std::endl;
365 // Add first 4 fillers
366 for (int j = 0; j < 4 && j < fillersToAdd; ++j) {
367 double fillerValueX = xyData(i, 0) + (j + 1) * (gap / (fillers + 1));
368 filledData.push_back(fillerValueX);
369 filledData_y.push_back(0.0);
370 }
371
372 // If there are more than 4 fillers, add the last 4 fillers
373 if (fillersToAdd > 4) {
374 /* mark a gap */
375 filledData.push_back(-1.0);
376 filledData_y.push_back(-1.0);
377 for (int j = fillers - 4; j < fillers; ++j) {
378 double fillerValueX = xyData(i, 0) + (j + 1) * (gap / (fillers + 1));
379 filledData.push_back(fillerValueX);
380 filledData_y.push_back(0.0);
381 }
382 }
383 if (fillersToAdd > 0) {
384 skipPrevGapCalculation = true;
385 }
386 }
387 // add last
388 filledData.push_back(xyData(xyData.numRows()-1,0));
389 filledData_y.push_back(xyData(xyData.numRows()-1,1));
390 // Transfer data from filledData to xyData
391 xyData.reinitialize(filledData.size(),2);
392 // Matrix xyData(filledData.size(),2);
393 for (size_t i = 0; i < filledData.size(); ++i) {
394 xyData(i, 0) = filledData[i];
395 xyData(i, 1) = filledData_y[i];
396 }
397
398 }

400 void Peakmodel::zeroInterpolation(Matrix& xyData) {
401 Matrix left_to_right = xyData;
402 zeroInterpolation_oneDirection(left_to_right);
403
404 // Flipping the matrix upside down
405 Matrix right_to_left = xyData;
406 for (size_t i = 0; i < right_to_left.numRows() / 2; i++) {
407 for (size_t j = 0; j < right_to_left.numCols(); j++) {
408 std::swap(right_to_left(i,j), right_to_left(right_to_left.numRows()-1-i, j));
409 }
410 }
411 zeroInterpolation_oneDirection(right_to_left);
412 // Flipping back
413 for (size_t i = 0; i < right_to_left.numRows() / 2; i++) {
414 for (size_t j = 0; j < right_to_left.numCols(); j++) {
415 std::swap(right_to_left(i,j), right_to_left(right_to_left.numRows()-1-i, j));
416 }
417 }
418 // Combining the results
419 for (size_t i = 0; i < xyData.numRows(); i++) {
420 xyData(i, 1) = std::max(left_to_right(i, 1), right_to_left(i, 1));
421 }

425 void Peakmodel::zeroInterpolation_oneDirection(Matrix& xyData) {
426 /* This function corrects zeros by interpolation considering the direct neighbors*/
427 size_t n = xyData.numRows();
428 // Checking for fronting zeros
429 size_t i = 0;
430 while (xyData(i, 1) == 0.)
431 {
432 i++;
433 if (i ==n) {
434 return;
435 }
436 }
437 if (i > 0) {
438 // There are fronting zeros, which will be corrected
439 for (int j = i-1; j >= 0; --j) {
440 xyData(j, 1) = xyData(j+1,1) / 2.;
441 }
442 }
443
444 // Checking the rest of the data (except last data point)
445 for (; i < n-1; i++) {
446 if (xyData(i,1) == 0.) {
447 xyData(i,1) = (xyData(i-1,1) + xyData(i+1,1)) / 2.;
448 }
449 }
450
451 // Checking the last data point
452 if (xyData(n-1,1) == 0.) {
453 xyData(n-1,1) = xyData(n-2,1) / 2.;
454 }

526 void Peakmodel::runRegression(
527 const std::pair<const dataID, const Matrix>& pair, const std::vector<int>& scales, const size_t k, const int s) {
528 /* This is the main function to perform the automated peak detection. */
529 // get key and data
530 dataID key = pair.first;
531 Matrix data = pair.second;
532 int peakID = 0;
533 // log transform
534 Matrix ylog = data.col(1).log(); // 1, i.e., first column (x) will be ignored
535
536 // preallocate conv Matrix that later will contain all regression coefficients
537 size_t n = ylog.numRows();
538 // check if scales fit with the data size
539 if (n > 2*scales[0]) {
540 size_t k2 = k;
541 int s2 = s;
542 std::vector<int> scales_tmp = scales;
543 int largest_scale = scales_tmp[scales_tmp.size()-1];
544 while (n < 2*largest_scale+1)
545 {
546 scales_tmp.pop_back();
547 k2 = scales_tmp.size();
548 s2 = 2*sum(scales_tmp);
549 largest_scale = scales_tmp[scales_tmp.size() - 1];
550 }
551
552 // Variables and Vectors
553 size_t N = k2*n - s2;
554 Matrix beta(4,N);
555 bool* fltrVec = new bool[N];
556 std::fill_n(fltrVec, N, true);
557
558 std::vector<int> xIndices(N);
559 std::vector<double> apex_positions(N);
560 std::vector<double> apex_positions_uncertainty(N);
561 std::vector<double> valley_positions(N);
562 std::vector<double> peakHeight(N);
563 std::vector<double> peakHeight_uncertainty(N);
564 std::vector<double> peakArea(N);
565 std::vector<double> peakArea_uncertainty(N);
566 std::vector<int> scaleVec = getScaleVec(scales_tmp, n, N);
567 std::vector<double> mse(N);
568 int currentIndex = 0;
569
570 // iterate through all scales
571 for (const auto& p : pseudoInverses) {
572 // check if data set is too small for full scan
573 if (p.second.numCols() < n) {
574 convolveP(beta, xIndices, p.second, ylog, currentIndex);
575 }
576 }
577
578 // Filtering Regression Results
579 // Vector of Indices that have passed the last filter
580 std::vector<int> idx1;
581 coefficientCriterion(N, fltrVec, scaleVec, apex_positions, valley_positions, xIndices, beta, idx1);
582
583 std::vector<int> idx2;
584 quadraticTermCriterion(N, fltrVec, scaleVec, xIndices, beta, ylog, mse, idx1, idx2);
585
586 idx1.clear();
587 parameterCriterion(N, fltrVec,peakHeight,peakHeight_uncertainty, peakArea, peakArea_uncertainty,scaleVec, beta, mse, idx2, idx1);
588
589 idx2.clear();
590 mergeRegressions(N, fltrVec, beta, scaleVec, xIndices, apex_positions, apex_positions_uncertainty, valley_positions, data.col(1), mse, idx1, idx2);
591
592 rescalePosition(N,fltrVec,data.col(0),apex_positions,apex_positions_uncertainty,idx2);
593
594 {
595 std::lock_guard<std::mutex> lock(mtx);
596 exportResults(beta, fltrVec, xIndices, N, key.sampleID,apex_positions,apex_positions_uncertainty,peakHeight,peakHeight_uncertainty,peakArea,peakArea_uncertainty,idx2,peakID);
597 }
598 delete[] fltrVec;
599 }

602 // Convolution
603 void Peakmodel::convolveP(
604 Matrix& beta,
605 std::vector<int>& xIndices,
606 const Matrix& P,
607 const Matrix& ylog,
608 int& currentIndex) {
609 size_t n = ylog.numRows();
610 size_t k = P.numCols();
611 size_t n_segments = n - k + 1;
612
613 Matrix p1 = ylog.convolveSymmetric(P.row(0));
614 Matrix p2 = ylog.convolveRotation(P.row(1));
615 Matrix p34 = ylog.convolveAntisymmetric(P.row(2));
616 int u = 0;
617 for (size_t j = 0; j < n_segments; j++)
618 {
619 beta(0,j + currentIndex) = p1(0, j);
620 beta(1,j + currentIndex) = p2(0, j);
621 beta(2,j + currentIndex) = p34(0,j);
622 beta(3,j + currentIndex) = p34(1,j);
623 xIndices[j + currentIndex] = u + k/2;
624 u++;
625 }
626 currentIndex += u;
627 }

1018 Matrix Peakmodel::calcJacobianMatrix_Area_half(
1019 const double b0,
1020 double b1,
1021 const double b2,
1022 const bool isLeft
1023 ) const {
1024 // precalculations:
1025 const double SQRTB2 = std::sqrt(-b2);
1026 const double EXP_B0 = std::exp(b0);
1027 const double B1_2_B2 = b1 / 2 / b2;
1028 const double EXP_B12 = std::exp(- b1 * B1_2_B2 / 2);
1029 double err_func_expr = 0.;
1030 /* the err_func_expr depends on the side */
1031 if (isLeft) {
1032 err_func_expr += std::erfc(b1/2/SQRTB2);
1033 } else {
1034 err_func_expr += 2 - std::erfc(b1/2/SQRTB2);
1035 }
1036
1037 // first term
1038 double J1 = EXP_B0 * EXP_B12 * err_func_expr / M_2_SQRTPI / SQRTB2;
1039 // second term
1040 double J2 = EXP_B0 / 2 / b2 - B1_2_B2 * J1;
1041 // third term
1042 double J3 = - J1 / 2 / b2 - B1_2_B2 * J2;
1043
1044 Matrix J(1,4);
1045 J(0,0) = J1;
1046 J(0,1) = J2;
1047 isLeft ? J(0,2) = J3 : J(0,3) = J3;
1048 return J;
1049 }

1052 const double b0,
1053 double b1,
1054 const double b2,
1055 const bool isLeft
1056 ) const {
1057 // Precalculation:
1058 const double SQRTB2 = std::sqrt(b2);
1059 const double EXP_B0 = std::exp(b0);
1060 const double B1_2_B2 = b1 / 2 / b2;
1061 const double EXP_B12 = std::exp(- b1 * B1_2_B2 / 2);
1062 double err_func_expr = 0.;
1063 /* the err_func_expr depends on the side */
1064 if (isLeft) {
1065 err_func_expr += erfi(b1/2/SQRTB2);
1066 } else {
1067 err_func_expr += -erfi(b1/2/SQRTB2);
1068 }
1069
1070 Matrix J(1,4);
1071 J(0,0) = EXP_B0 * EXP_B12 / M_2_SQRTPI / SQRTB2;
1072 J(0,1) = EXP_B0 / 2 / b2 - B1_2_B2 * J(0,0);
1073 if (isLeft) {
1074 J(0,2) = - J(0,0) / 2 / b2 - B1_2_B2 * J(0,1) ;
1075 } else {
1076 J(0,3) = - J(0,0) / 2 / b2 - B1_2_B2 * J(0,1) ;
1077 }

1149 ) {

1184 // std::cout << idx_ref.size() << std::endl;

1198

1200 yhat(j,0) = std::exp(yhat(j,0));

1217 // these peaks cannot be distinguished and are therefore similar
1218 idx_candidates.insert(idx);
1219 n_Candidates++;

1221 if (mse_exp[col] == 0) {

1227 }
1228 mse_exp[col] = calcMse(yhat, y);
1229 }
1230 }
1231 }
1232 }
1233 // check if there are candidates for mse_comparison:
1234 if (n_Candidates > 0) {
1235 // There are similar peaks: start survival of the fittest
1236 // calculate the average of the candidates' mse_exp
1237 double mse_exp_ref = 0;
1238 for (int i : idx_candidates) {
1239 mse_exp_ref += (mse_exp[i] / n_Candidates);
1240 }
1241 if (mse_exp[col] < mse_exp_ref) {
1242 // new peak is better than former candidates
1243 idx_ref.insert(col);
1244 // delete old ones from list
1245 for (int i : idx_candidates) {
1246 idx_ref.erase(i);
1247 }
1248 }
1249 } else {
1250 // No similar peaks: add new peak to the list
1251 idx_ref.insert(col);
1252 }
1253 }
1254 }
1255 for (int idx : idx_ref) {
1256 idx2.push_back(idx);
1257 }
1258 delete[] mse_exp;
1259 delete[] var_apex;
